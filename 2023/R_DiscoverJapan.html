Data <- recipe(
  ~.,
  Raw
  ) |> 
  step_filter(Tenure >= 0) |> 
  step_filter(Size >= 15) |> 
  step_filter(Size <= 110) |> 
  step_filter(Tenure <= 50) |> 
  step_filter(Youseki <= 700) |> 
  step_filter(RoomNum <= 4) |> 
  step_dummy(
    District,
    one_hot = TRUE) |> 
  step_scale(
    Tenure,
    Size,
    Youseki,
    Kenpei,
    RoomNum,
    Quartely
  ) |> 
  prep() |> 
  bake(new_data = NULL)

X <- Data |> 
  select(
    -Price,
    -After
  )

NameX <- names(X)
names(X) <- str_c("X",1:ncol(X))

Y <- Data$Price

D <- Data$After

    PredLinearAIPW$response[Group == 3], 
    probs = q)

    PredR$response[Group == 3], 
    probs = q)
  
  TempAIPW <- Q_AIPW +

Data <- Data[sample(1:nrow(Data),200),]

X <- recipe(
  ~ .,
  Data |> 
    select(
      -Price,
      -After
    )
  ) |> 
  step_interact(
    ~ Size:all_predictors()
  ) |> 
  step_normalize(
    all_predictors()
  ) |> 
  prep() |> 
  bake(
    new_data = NULL
  )

Y <- Data$Price|> log()

D <- Data$After

RegLM <- lrn(
  "regr.lm",
  id = "OLS"
  )

LASSO <- lrn("regr.glmnet") |> 
  lts()

LASSO <- 
  LASSO |> 
  as_learner()

LASSO <- AutoTuner$new(
  learner = LASSO,
  resampling = CV,
  terminator = Terminal,
  tuner = Tuner
  )

LASSO$id <- "LASSO"

Benchmark

Task <- as_task_regr(
  X |> 
    mutate(Y = Y),
  target = "Y"
  )

Design <- benchmark_grid(
  tasks = Task,
  learners = list(
    LASSO,
    RegLM
  ),
  resamplings = CV
)

lgr::get_logger("mlr3")$set_threshold("error") # Errorのみを表示

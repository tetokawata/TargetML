Y <- Data$Price
W <- Data$Reform
X <- Data |> 
  select(
    -Price,
    -Reform
    )

Group <- sample(0:1,length(Y),replace = TRUE)

CIfun <- conformalIte(
  X[Group == 0,], 
  Y[Group == 0], 
  W[Group == 0], 
  alpha = 0.1, 
  algo = "nest", 
  exact = FALSE, 
  type = "CQR",
  quantiles = c(0.05, 0.95), 
  outfun = "quantRF", 
  useCV = FALSE
  )
Loading required namespace: grf
Result <- X[Group == 1,] |> 
  mutate(
    Lower = CIfun(X[Group == 1,])[1],
    Upper = CIfun(X[Group == 1,])[2]
  )

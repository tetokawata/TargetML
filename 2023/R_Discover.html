DetectOutLinear <- function(x){
  check <- abs(x - mean(x)) <= 3*sd(x)
  return(check)
}

Data <- recipe(
  ~.,
  Raw
  ) |> 
  step_filter(
    DetectOutLinear(Raw$Tenure) &
      DetectOutLinear(Raw$Size) &
      DetectOutLinear(Raw$DistanceStation) &
      DetectOutLinear(Raw$Youseki) &
      DetectOutLinear(Raw$Kenpei) &
      DetectOutLinear(Raw$RoomNum)
    ) |> 
  step_filter(
    Tenure >= 0
  ) |> 
  step_scale(
    Tenure,
    DistanceStation,
    Size,
    Youseki,
    Kenpei,
    RoomNum,
    Quartely
  ) |> 
  prep() |> 
  bake(new_data = NULL)

Est <- function(q){
  Q_AIPW <- quantile(
    PredAIPW$response[Group == 3], 
    probs = q)
  Q_LinearAIPW <- quantile(
    PredLinearAIPW$response[Group == 3], 
    probs = q)
  
  TempAIPW <- Q_AIPW + 
    (1/q)*if_else(
      PredAIPW$response[Group == 3] <= Q_AIPW,
      ScoreAIPW[Group == 3]- Q_AIPW,
      0
      )
  TempLinearAIPW <- Q_LinearAIPW + 
    (1/q)*if_else(
      PredLinearAIPW$response[Group == 3] <= Q_LinearAIPW,
      ScoreAIPW[Group == 3]- Q_LinearAIPW,
      0
      )
  TempResult <- estimatr::lm_robust(
    TempAIPW ~ 1
    ) |> 
    generics::tidy() |> 
    mutate(
      Quantile = q,
      Method = "AIPW with SL"
      ) |> 
    bind_rows(
      estimatr::lm_robust(
    TempLinearAIPW ~ 1
    ) |> 
    generics::tidy() |> 
    mutate(
      Quantile = q,
      Method = "AIPW with OLS"
      )
    )
  return(TempResult)
}

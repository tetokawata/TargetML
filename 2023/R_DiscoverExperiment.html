Data <- recipe(
  ~.,
  Raw
  ) |> 
  step_filter(Tenure >= 0) |> 
  step_filter(Size >= 15) |> 
  step_filter(Size <= 110) |> 
  step_filter(Tenure <= 50) |> 
  step_filter(DistanceStation <= 20) |> 
  step_filter(Youseki <= 700) |> 
  step_filter(RoomNum <= 4) |> 
  step_dummy(
    District,
    one_hot = TRUE) |> 
  step_scale(
    Tenure,
    DistanceStation,
    Size,
    Youseki,
    Kenpei,
    RoomNum,
    Quartely
  ) |> 
  prep() |> 
  bake(new_data = NULL)

X <- Data |>

AIPW base

ScoreY1 <- PredY1$response + 
  (D/PredD$response)*
  (Y - PredY1$response)

ScoreY0 <- PredY0$response + 
  ((1-D)/(1-PredD$response))*
  (Y - PredY0$response)

Est <- function(q){
  Q_AIPW <- quantile(
    PredAIPW$response[Group == 3], 
    probs = q)
  Q_LinearAIPW <- quantile(
    PredLinearAIPW$response[Group == 3], 
    probs = q)
  Q_R <- quantile(
    PredR$response[Group == 3], 
    probs = q)
  Q_ATT <- quantile(
    PredATT$response[Group == 3], 
    probs = q)
  
  TempAIPW <- Q_AIPW + 
    (1/q)*if_else(
      PredAIPW$response[Group == 3] <= Q_AIPW,
      ScoreAIPW[Group == 3]- Q_AIPW,
      0
      )
  TempLinearAIPW <- Q_LinearAIPW + 
    (1/q)*if_else(
      PredLinearAIPW$response[Group == 3] <= Q_LinearAIPW,
      ScoreAIPW[Group == 3]- Q_LinearAIPW,
      0
      )
  TempR <- Q_R +
    (1/q)*if_else(
      PredR$response[Group == 3] <= Q_R,
      ScoreAIPW[Group == 3]- Q_R,
      0
      )
  TempATT <- Q_ATT +
    (1/q)*if_else(
      PredATT$response[Group == 3] <= Q_ATT,
      ScoreAIPW[Group == 3]- Q_ATT,
      0
      )
  TempResult <- estimatr::lm_robust(
    TempAIPW ~ 1
    ) |> 
    generics::tidy() |> 
    mutate(
      Quantile = q,
      Method = "AIPW with SL"
      ) |> 
    bind_rows(
      estimatr::lm_robust(
    TempLinearAIPW ~ 1
    ) |> 
    generics::tidy() |> 
    mutate(
      Quantile = q,
      Method = "AIPW with OLS"
      )
    ) |> 
    bind_rows(
      estimatr::lm_robust(
    TempR ~ 1
    ) |> 
    generics::tidy() |> 
    mutate(
      Quantile = q,
      Method = "R"
      )
    ) |> 
    bind_rows(
      estimatr::lm_robust(
    TempATT ~ 1
    ) |> 
    generics::tidy() |> 
    mutate(
      Quantile = q,
      Method = "ATT"
      )
    )
  return(TempResult)
}
